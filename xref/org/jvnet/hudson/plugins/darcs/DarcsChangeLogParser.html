33  /**
34   * Parses the output of darcs log.
35   * 
36   * @author Sven Strittmatter <ich@weltraumschaf.de>
37   */
38  public class DarcsChangeLogParser  extends ChangeLogParser {
39      private static final Logger logger = Logger.getLogger(DarcsChangeLogParser.class.getName());
40      
41      @Override
42      public ChangeLogSet<? extends Entry> parse(AbstractBuild build, File changelogFile) throws IOException, SAXException {
43          XMLReader       xr      = XMLReaderFactory.createXMLReader();
44          DarcsSaxHandler handler = new DarcsSaxHandler();
45          FileReader      r       = new FileReader(changelogFile);
46  
47  	xr.setContentHandler(handler);
48  	xr.setErrorHandler(handler); 
49          xr.parse(new InputSource(r));
50  
51          return new DarcsChangeSetList(build, handler.getChangeSets());
52      }
