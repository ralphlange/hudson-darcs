97      /**
98       * Writes the cahngelog of the last numPatches to the changeLog file.
99       * 
100      * @param launcher
101      * @param numPatches
102      * @param workspace
103      * @param changeLog
104      * @throws InterruptedException
105      */
106     private void getLog(Launcher launcher, int numPatches, FilePath workspace, File changeLog) throws InterruptedException {
107         try {
108             int ret;
109             ByteArrayOutputStream baos = new ByteArrayOutputStream();
110             ProcStarter proc = launcher.launch()
111                                        .cmds(getDescriptor().getDarcsExe(), "changes", "--xml-output", "--last=" + numPatches, "--summary")
112                                        .envs(EnvVars.masterEnvVars)
113                                        .stdout(baos)
114                                        .pwd(workspace);
115             ret = proc.join();
116 
117             if (ret != 0) {
118                 logger.log(Level.WARNING, "darcs changes  --last=" + numPatches + " returned {0}", ret);
119             } else {
120                 FileOutputStream fos = new FileOutputStream(changeLog);
121                 fos.write(baos.toByteArray());
122                 fos.close();
123             }
124         } catch (IOException e) {
125             StringWriter w = new StringWriter();
126             e.printStackTrace(new PrintWriter(w));
127             logger.log(Level.WARNING, "Failed to poll repository: ", e);
128         }
129     }

150     @Override
151     public boolean checkout(AbstractBuild<?, ?> build, Launcher launcher, FilePath workspace, BuildListener listener, File changelogFile) throws IOException, InterruptedException {
152         boolean existsRepoinWorkspace = workspace.act(new FileCallable<Boolean>() {
153 
154             private static final long serialVersionUID = 1L;
155 
156             public Boolean invoke(File ws, VirtualChannel channel) throws IOException {
157                 File file = new File(ws, "_darcs");
158                 return file.exists();
159             }
160         });
161 
162         if (existsRepoinWorkspace && !isClean()) {
163             return pullRepo(build, launcher, workspace, listener, changelogFile);
164         } else {
165             return getRepo(build, launcher, workspace, listener, changelogFile);
166         }
167     }

169     /**
170      * Counts the patches in an repo.
171      * 
172      * @param build
173      * @param launcher
174      * @param workspace
175      * @param listener
176      * @return int
177      * @throws InterruptedException
178      * @throws IOException
179      */
180     private int countPatches(AbstractBuild<?, ?> build, Launcher launcher, FilePath workspace, BuildListener listener) throws InterruptedException, IOException {
181         ByteBuffer baos = new ByteBuffer();
182         ProcStarter proc = launcher.launch()
183                                    .cmds(getDescriptor().getDarcsExe(), "changes", "--count", "--repodir=" + workspace)
184                                    .envs(build.getEnvironment(listener))
185                                    .stdout(baos);
186         if (proc.join() != 0) {
187             listener.error("Failed to count patches in workspace repo!");
188             return 0;
189         }
190         
191         return Integer.parseInt(baos.toString().trim());
192     }

194     /**
195      * Pulls all patches from a remote repo in the workspace repo.
196      *
197      * @param build
198      * @param launcher
199      * @param workspace
200      * @param listener
201      * @param changelogFile
202      * @return boolean
203      * @throws InterruptedException
204      * @throws IOException
205      */
206     private boolean pullRepo(AbstractBuild<?, ?> build, Launcher launcher, FilePath workspace, BuildListener listener, File changelogFile) throws InterruptedException, IOException {
207         logger.log(Level.INFO, "Pulling repo from: {0}", source);
208         int preCnt = 0, postCnt = 0;
209 
210         try {
211             preCnt = countPatches(build, launcher, workspace, listener);
212             logger.log(Level.INFO, "Count of patches pre pulling is {0}", preCnt);
213             ProcStarter proc = launcher.launch()
214                                        .cmds(getDescriptor().getDarcsExe(), "pull", source, "--all", "--repodir=" + workspace )
215                                        .envs(build.getEnvironment(listener))
216                                        .stdout(listener.getLogger())
217                                        .pwd(workspace);
218 
219             if (proc.join() != 0) {
220                 listener.error("Failed to pull");
221 
222                 return false;
223             }
224 
225             postCnt = countPatches(build, launcher, workspace, listener);
226             logger.log(Level.INFO, "Count of patches post pulling is {0}", preCnt);
227             getLog(launcher, postCnt - preCnt, workspace, changelogFile);
228         } catch (IOException e) {
229             listener.error("Failed to pull");
230             
231             return false;
232         }
233 
234         return true;
235     }

237     /**
238      * Gets a fresh copy of a remote repo.
239      *
240      * @param build
241      * @param launcher
242      * @param workspace
243      * @param listener
244      * @param changelogFile
245      * @return boolean
246      * @throws InterruptedException
247      */
248     private boolean getRepo(AbstractBuild<?, ?> build, Launcher launcher, FilePath workspace, BuildListener listener, File changelogFile) throws InterruptedException {
249         logger.log(Level.INFO, "Getting repo from: {0}", source);
250         
251         try {
252             workspace.deleteRecursive();
253         } catch (IOException e) {
254             e.printStackTrace(listener.error("Failed to clean the workspace"));
255             return false;
256         }
257 
258         ArgumentListBuilder args = new ArgumentListBuilder();
259         args.add(getDescriptor().getDarcsExe(), "get");
260         args.add(source, workspace.getRemote());
261 
262         try {
263             ProcStarter proc = launcher.launch()
264                                        .cmds(args)
265                                        .envs(build.getEnvironment(listener))
266                                        .stdout(listener.getLogger());
267             
268             if (proc.join() != 0) {
269                 listener.error("Failed to get " + source);
270 
271                 return false;
272             }
273         } catch (IOException e) {
274             e.printStackTrace(listener.error("Failed to get " + source));
275             
276             return false;
277         }
278 
279         return createEmptyChangeLog(changelogFile, listener, "changelog");
280     }

355                        ProcStarter proc  = launcher.launch()
356                                                    .cmds(getDarcsExe(), "--version")
357                                                    .stdout(baos);
358                        if (proc.join() == 0) {
359                           return FormValidation.ok();
360                        } else {
361                           return FormValidation.warning("Could not locate the executable in path");
